86def __draw_header( 

87 username: str, 

88 user_tag: str, 

89 user_pic: str, 

90 graphic_size: List[int], 

91 img: Image.Image, 

92 d_interface: ImageDraw.ImageDraw, 

93 coords: List[int], 

94 margin: float, 

95 font: ImageFont.FreeTypeFont, 

96 color: str, 

97) -> int: 

98 """Draw the graphic's header (username, user tag and, if specified, profile picture). 

99 

100 Parameters 

101 ---------- 

102 username : str 

103 Tweet's username. 

104 user_tag : str 

105 Tweet's user tag/handle. 

106 user_pic : str 

107 Path to user's profile picture. 

108 graphic_size : List[int] 

109 Dimensions of the graphic. 

110 img : Image 

111 PIL Image used to draw the graphic. 

112 d_interface : ImageDraw 

113 Drawing interface for the Image. 

114 coords : List[int] 

115 Coordinates at which to start drawing. 

116 margin : float 

117 Vertical margin between text lines. 

118 font : ImageFont.FreeTypeFont 

119 Font used for the graphic's header. 

120 color : str 

121 Text color. 

122 

123 Returns 

124 ------- 

125 int 

126 The current Y coordinate. 

127 """ 

128 x = int(coords[0]) 

129 y = int(coords[1]) 

130 

131 # If a profile picture was not specified, draw the username and user tag 

132 if user_pic == "": 

133 # Draw the username 

134 user_name = wrap(username, 38) 

135 for line in user_name: 

136 d_interface.text((x, y), line, font=font, fill=color) 

137 y += font.size + margin 

138 # Draw the user handle 

139 d_interface.text((x, y), user_tag, font=font, fill=color) 

140 y += int(font.size + margin * 1.5) 

141 

142 # Otherwise, draw the profile picture too 

143 else: 

144 # Process the profile picture and draw it 

145 user_pic_processed = process_pic(graphic_size, user_pic) 

146 img.paste(user_pic_processed, (x, y), mask=user_pic_processed) 

147 # Due to the presence of the profile picture, the horizontal\ 

148 # coordinate for the rest of the header is updated 

149 x = int(coords[0] + user_pic_processed.size[0] + margin) 

150 # Draw the username 

151 user_name = wrap(username, 19) 

152 for line in user_name: 

153 d_interface.text((x, y), line, font=font, fill=color) 

154 y += font.size + margin 

155 # Draw the user tag 

156 d_interface.text((x, y), user_tag, font=font, fill=color) 

157 y += int(font.size + margin * 1.5) 

158 

159 # Return the current vertical coordinate 

160 return y

163def __get_initial_coordinates( 

164 img_size: List[int], dimensions: Dict[str, List[int]] 

165) -> Tuple[int]: 

166 """Calculate the initial X and Y coordinates at which to start drawing. 

167 

168 Parameters 

169 ---------- 

170 img_size : List[int] 

171 Width and height of the graphic. 

172 dimensions : Dict[str, List[int]] 

173 Dictinary that contains the size of the header (without profile picture) and the tweet text. 

174 

175 Returns 

176 ------- 

177 Tuple[int] 

178 Initial coordinates to start drawing at. 

179 """ 

180 # Get header and tweet text dimensions 

181 header_width, header_height = dimensions["header"] 

182 text_width, text_height = dimensions["text"] 

183 

184 # Horizontal coordinate to draw at (centered) 

185 x = (img_size[0] - text_width) // 2 

186 # Initial vertical coordinate to draw at (centered) 

187 content_height = header_height + text_height 

188 y = (img_size[1] - content_height) // 2 

189 

190 return (x, y)

193def create_tweet( 

194 tweet_info: TweetInfo, 

195 graphic_settings: GraphicSettings, 

196 default_settings_format: DefaultFormats = DefaultFormats.CUSTOM.value, 

197 save_dir: Optional[str] = "", 

198) -> None: 

199 """Create a tweet graphic. 

200 

201 Parameters 

202 ---------- 

203 tweet_info : TweetInfo 

204 Dictionary with the necessary information about the tweet. 

205 graphic_settings : GraphicSettings 

206 Dictionary with the settings needed to draw the graphic. 

207 default_settings_format : DefaultFormats, optional 

208 Default graphic settings option chosen, by default DefaultFormats.CUSTOM.value 

209 save_dir : Optional[str], optional 

210 Directory in which to save the graphic., by default "" 

211 """ 

212 # Validate the tweet info 

213 t_info = validate_tweet_info(tweet_info) 

214 # Use the graphic settings passed (either custom or default) 

215 g_settings = __choose_graphic_settings(graphic_settings, default_settings_format) 

216 

217 # Get the tweet info received 

218 user_name = tweet_info["user_name"] 

219 user_tag = tweet_info["user_tag"] 

220 tweet_text = tweet_info["tweet_text"] 

221 user_pic = tweet_info["user_pic"] 

222 

223 # Get the graphic settings received 

224 font_family = g_settings["font_family"] 

225 font_size_text = g_settings["font_size_text"] 

226 font_size_header = g_settings["font_size_header"] 

227 # Set up the fonts based on settings 

228 font_header = ImageFont.truetype(font_family, font_size_header, encoding="utf-8") 

229 font_text = ImageFont.truetype(font_family, font_size_text, encoding="utf-8") 

230 

231 # Size of the graphic 

232 img_size = g_settings["size"] 

233 # Vertical margin in between lines 

234 margin_bottom = g_settings["margin_bottom"] 

235 background_color = g_settings["color_scheme"][0] 

236 text_color = g_settings["color_scheme"][1] 

237 chars_limit = g_settings["wrap_limit"] 

238 

239 # Dict with size of header and size of text 

240 content_dims = calculate_content_dimensions(tweet_info, g_settings) 

241 # Calculate the inital drawing coordinates 

242 x, y = __get_initial_coordinates(img_size, content_dims) 

243 

244 # Create what will be the final image 

245 img = Image.new("RGBA", (img_size[0], img_size[1]), color=background_color) 

246 # Create the drawing interface 

247 draw = ImageDraw.Draw(img) 

248 

249 # Draw the header text (and update the vertical coordinate to be where\ 

250 # the header finishes) 

251 y = __draw_header( 

252 user_name, 

253 user_tag, 

254 user_pic, 

255 img_size, 

256 img, 

257 draw, 

258 [x, y], 

259 margin_bottom, 

260 font_header, 

261 text_color, 

262 ) 

263 

264 # Split the tweet text into lines 

265 text_wrapped = wrap(tweet_text, chars_limit) 

266 # Draw the tweet text 

267 for line in text_wrapped: 

268 draw.text((x, y), line, font=font_text, fill=text_color) 

269 y += font_text.size + margin_bottom 

270 

271 save_name = f"{tweet_info['tweet_name']}.png" 

272 save_name = path.join(save_dir, save_name) 

273 img.save(save_name)
