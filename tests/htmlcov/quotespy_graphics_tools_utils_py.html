11def __load_quotes_txt(file_path: str) -> List[Tuple[str]]: 

12 """Scrape quotes from a given TXT file. Titles need to be wrapped 

13 by square brackets ([]) and the respective quote needs to come in 

14 the next line. 

15 

16 Parameters 

17 ---------- 

18 file_path : str 

19 Path to the .txt file with lyrics/quotes. 

20 

21 Returns 

22 ------- 

23 List[Tuple[str]] 

24 List of tuples that contain the title and text of each quote loaded. 

25 """ 

26 # Load the source file contents as a single string 

27 with open(file_path, "r", encoding="utf-8") as source_file: 

28 contents = "".join(source_file.readlines()) 

29 

30 # Get all the titles 

31 pattern_titles = r"\[(.*?)\]" 

32 titles = re.findall(pattern_titles, contents) 

33 # Get all the quotes 

34 pattern_quotes = r"^([^\[].*?[^\]])$" 

35 quotes = re.findall(pattern_quotes, contents, flags=re.MULTILINE) 

36 

37 # Validate the loaded titles and quotes 

38 __validate_text_loaded(titles, quotes) 

39 

40 # Pair the titles with the respective quotes in two-item lists 

41 titles_quotes = list(zip(titles, quotes)) 

42 

43 return titles_quotes

82def __update_title_counts(quotes: List[Tuple[str]]) -> Dict[str, str]: 

83 """Given a list of lists of titles and quotes loaded from a .txt file, update the titles with the respective frequencies. 

84 

85 Parameters 

86 ---------- 

87 quotes : List[Tuple[str]] 

88 List of tuples that contain the title and quote of each graphic. 

89 

90 Returns 

91 ------- 

92 Dict[str, str] 

93 Dictionary that maps titles to the corresponding lyrics/quote. 

94 """ 

95 # Freqs of each unique quote 

96 title_freqs = {} 

97 # Dictionary with the updated titles mapped to the corresponding quotes 

98 updated_quotes = {} 

99 

100 # Loop through the loaded quotes to update titles with their frequencies 

101 for quote in quotes: 

102 title = quote[0] 

103 text = quote[1] 

104 

105 # If this quote title has been seen before, update the title with its current frequency 

106 if title in title_freqs: 

107 # Update the title frequency 

108 title_freqs[title] += 1 

109 

110 # Update the title with its current frequency 

111 updated_title = f"{title} {str(title_freqs[title])}" 

112 

113 # Add the updated quote information to the dictionary 

114 updated_quotes[updated_title] = text 

115 

116 # If this is the first time seeing the quote, simply use it as is 

117 else: 

118 title_freqs[title] = 1 

119 updated_quotes[title] = text 

120 

121 return updated_quotes
